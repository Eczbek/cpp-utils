## Example
```cpp
import std;
import xieite;

int main() {
    auto litres = xieite::units::Litre(5);
    auto gallons = static_cast<xieite::units::Gallon>(litres);

    std::println("{} L", litres.value);
    std::println("{} gal", gallons.value);
}
```
Possible output:
```
5 L
1.099845787 gal
```
